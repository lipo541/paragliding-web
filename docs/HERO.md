## рЃќрЃЮрЃњрЃљрЃЊрЃў рЃЏрЃўрЃЏрЃЮрЃ«рЃўрЃџрЃЋрЃљ

Hero рЃАрЃћрЃЦрЃфрЃўрЃљ рЃгрЃљрЃарЃЏрЃЮрЃљрЃЊрЃњрЃћрЃюрЃА **рЃЊрЃўрЃюрЃљрЃЏрЃўрЃБрЃа рЃАрЃџрЃљрЃўрЃЊрЃћрЃарЃА**, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃЏрЃЌрЃџрЃўрЃљрЃюрЃљрЃЊ рЃўрЃЏрЃљрЃарЃЌрЃћрЃЉрЃљ рЃљрЃЊрЃЏрЃўрЃю рЃърЃљрЃюрЃћрЃџрЃўрЃЊрЃљрЃю. рЃАрЃўрЃАрЃбрЃћрЃЏрЃљ рЃАрЃљрЃерЃБрЃљрЃџрЃћрЃЉрЃљрЃА рЃљрЃФрЃџрЃћрЃЋрЃА Super Admin-рЃА:

- рЃЊрЃљрЃљрЃЏрЃљрЃбрЃЮрЃА рЃбрЃћрЃЦрЃАрЃбрЃў рЃАрЃљрЃЏ рЃћрЃюрЃљрЃќрЃћ (рЃЦрЃљрЃарЃЌрЃБрЃџрЃў, рЃўрЃюрЃњрЃџрЃўрЃАрЃБрЃарЃў, рЃарЃБрЃАрЃБрЃџрЃў)
- рЃљрЃбрЃЋрЃўрЃарЃЌрЃЮрЃА рЃАрЃБрЃарЃљрЃЌрЃћрЃЉрЃў Light рЃЊрЃљ Dark рЃарЃћрЃЪрЃўрЃЏрЃўрЃАрЃЌрЃЋрЃўрЃА рЃфрЃљрЃџ-рЃфрЃљрЃџрЃЎрЃћ
- рЃЊрЃљрЃљрЃЏрЃљрЃбрЃЮрЃА рЃдрЃўрЃџрЃљрЃЎрЃћрЃЉрЃў рЃАрЃ«рЃЋрЃљрЃЊрЃљрЃАрЃ«рЃЋрЃљ рЃЏрЃЮрЃЦрЃЏрЃћрЃЊрЃћрЃЉрЃћрЃЉрЃўрЃЌ
- рЃњрЃљрЃЊрЃљрЃљрЃџрЃљрЃњрЃЮрЃА рЃАрЃџрЃљрЃўрЃЊрЃћрЃЉрЃў Drag-and-Drop рЃЏрЃћрЃЌрЃЮрЃЊрЃўрЃЌ

### рЃдрЃўрЃџрЃљрЃЎрЃўрЃА Click Handler (Frontend)

```typescript
const handleButtonClick = (button: HeroSlideButton) => {
  switch (button.action_type) {
    case 'link':
      // URL-рЃќрЃћ рЃњрЃљрЃЊрЃљрЃАрЃЋрЃџрЃљ
      // рЃљрЃ«рЃљрЃџ рЃбрЃљрЃЉрЃерЃў рЃЌрЃБ open_in_new_tab === true рЃљрЃю рЃњрЃљрЃарЃћ URL-рЃўрЃљ
      if (button.open_in_new_tab || isExternalUrl(button.action_url)) {
        window.open(button.action_url, '_blank', 'noopener,noreferrer');
      } else {
        router.push(button.action_url);
      }
      break;
    case 'contact':
      // рЃњрЃљрЃЊрЃљрЃАрЃЋрЃџрЃљ /{locale}/contact
      router.push(`/${locale}/contact`);
      break;
    case 'specialist':
      // рЃњрЃљрЃЊрЃљрЃАрЃЋрЃџрЃљ /{locale}/specialists/{specialist_id}
      router.push(`/${locale}/specialists/${button.specialist_id}`);
      break;
    case 'practice':
      // рЃњрЃљрЃЊрЃљрЃАрЃЋрЃџрЃљ /{locale}/practices/{practice_id}
      router.push(`/${locale}/practices/${button.practice_id}`);
      break;
    case 'company':
      // рЃњрЃљрЃЊрЃљрЃАрЃЋрЃџрЃљ /{locale}/companies/{company_id}
      router.push(`/${locale}/companies/${button.company_id}`);
      break;
  }
}
```
